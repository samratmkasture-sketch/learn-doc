# Program to display the Fibonacci sequence up to n-th term

def genFibonacci(n=10):
  fiboList=[0,1]
  for i in range(2,n):
     fiboList.append(fiboList[i-1]+fiboList[i-2])
     
  return fiboList
    

print(len(genFibonacci(20)))
